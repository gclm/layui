# fsonCheckBox

项目介绍
layui 拓展插件,二级复选框.能力有限，望有能力者继续改善



安装教程



使用说明
<div id='test1'></div>
<button type='button' id='btn'>button</button>

layui.config({
  base: '../js/layui_exts/' //配置 layui 第三方扩展组件存放的基础目录
}).extend({
  fsonCheckBox: 'fsonCheckBox/fsonCheckBox' //定义该组件模块名
}).use(['layer', 'fsonCheckBox'], function () {
        var layer = layui.layer
            , $ = layui.$
            ,fsonCheckBox = layui.fsonCheckBox;
            ,fson = fsonCheckBox.render({
                elem: '#test1'
                , url: "{:url('admin/department/framework')}"
                // 默认值
                , value: [
                    {id: 1, child: [22, 45, 108]}
                    , {id: 3, child: [4, 5, 6, 7]}
                    , {id: 5, child: [11, 22, 33, 44, 55]}
                ]
            });
        $('#btn').click(function () {
            layer.alert(JSON.stringify(fson.config.value))
        })
});
