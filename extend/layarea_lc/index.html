//配置插件目录
        layui.config({
            base: './mods/'
            , version: '1.0'
        });
        //一般直接写在一个js文件中
        layui.use(['layer', 'form', 'layarea_lc'], function () {
            var layer = layui.layer
                , form = layui.form
                , layarea_lc = layui.layarea_lc;

            layarea_lc.render({
                elem: '#area-picker',
		//不填均为code类型
                name:'name',
                change: function (res) {
                    //选择结果
                    console.log(res);
                }
            });
        });
