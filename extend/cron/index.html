文档

1. 将 cron 文件夹放到扩展组件目录

2. 在使用页面引入模块 
layui
    .config({
      base: "../static/layui_exts/", // 扩展组件跟路径
    })
    .extend({
      cron: "cron/cron" // 扩展组件
    })
    .use(["cron"], function () {
      var $ = layui.$,
        cron = layui.cron;
        cron.render({
        elem: "#cron", // 绑定元素
	run: "../static/json/run.json", // 获取最近运行时间的接口
	done: function(cronStr) {
	    console.log(cronStr);
	}
      });
    });
